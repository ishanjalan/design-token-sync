Usage

Installation

Add DesignTokens.swift to your Xcode project.

SwiftUI Example

import SwiftUI

struct ContentView: View {
    var body: some View {
        VStack(spacing: DesignTokens.Spacing.spacing16) {
            Text("Hello, World!")
                .font(DesignTokens.Typography.heading1.font)
                .foregroundColor(.textPrimary)
            
            Button("Continue") {
                // Action
            }
            .foregroundColor(.staticWhite)
            .padding(DesignTokens.Spacing.spacing16)
            .background(.fillBrandPrimary)
            .cornerRadius(DesignTokens.Radius.radius8)
        }
        .padding(DesignTokens.Spacing.spacing24)
        .background(.backgroundBase)
    }
}

Direct Token Access

// Using nested enums
let textColor = DesignTokens.Colors.Text.primary.color
let spacing = DesignTokens.Spacing.spacing16
let radius = DesignTokens.Radius.radius8

// Using convenience extensions
let buttonColor = Color.fillBrandPrimary
let backgroundColor = Color.backgroundBase

ColorToken Struct

All color tokens use the ColorToken struct which automatically handles light/dark mode:

struct ColorToken {
    let light: Color
    let dark: Color
    
    var color: Color {
        // Automatically switches based on system theme
    }
}
