Usage

CSS

Import the CSS files in your project:

@import url('tokens/colors.css');
@import url('tokens/spacing.css');
@import url('tokens/radius.css');
@import url('tokens/typography.css');

Use the CSS variables:

.button {
  color: var(--text-primary);
  background: var(--fill-brand-primary);
  padding: var(--spacing-16);
  border-radius: var(--radius-8);
}

TypeScript

Import the TypeScript tokens:

import { colors, spacing, radius } from './tokens/colors';

const buttonStyle = {
  color: colors.text.primary,
  padding: `${spacing[16]}px`,
  borderRadius: `${radius[8]}px`,
};

React Example

import { colors, spacing } from './tokens/colors';

function Button({ children }) {
  return (
    <button style={{
      backgroundColor: colors.fill.brand.primary,
      color: colors.text.primary,
      padding: `${spacing[12]}px ${spacing[20]}px`,
      borderRadius: `${radius[8]}px`,
    }}>
      {children}
    </button>
  );
}
